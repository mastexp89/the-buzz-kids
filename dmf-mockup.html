DATA MODEL

One new migration

CREATE TABLE festivals (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name        text NOT NULL,
  slug        text NOT NULL UNIQUE,
  start_date  date NOT NULL,
  end_date    date NOT NULL,
  hero_image_url text,
  primary_color  text DEFAULT '#e91e63',
  sponsor_text   text,
  ticket_url     text,
  description    text,
  published      boolean DEFAULT false,
  created_at     timestamptz DEFAULT now()
);

CREATE TABLE festival_venues (
  festival_id  uuid REFERENCES festivals(id) ON DELETE CASCADE,
  venue_id     uuid REFERENCES venues(id)    ON DELETE CASCADE,
  sort_order   int DEFAULT 0,
  PRIMARY KEY (festival_id, venue_id)
);

-- Public can read published festivals
ALTER TABLE festivals ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Read published festivals" ON festivals
  FOR SELECT USING (published);
ALTER TABLE festival_venues ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Read festival venues" ON festival_venues
  FOR SELECT USING (true);
